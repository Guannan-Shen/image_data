PyTorch: convolutional neural network for cat recognition¶

A simple convolutional neural network is trained to predict the class labels (cat or non-cat) by minimizing cross-entropy loss.

This implementation is based on deeplearning.ai (https://www.deeplearning.ai/) and PyTorch tutorial (https://pytorch.org/tutorials/index.html).

def load_data():

    return train_set_x_orig, train_set_y_orig, test_set_x_orig, test_set_y_orig, classes

import torch
import torch.nn as nn
import torch.nn.functional as F

# Define the network
class Model(nn.Module):
    def __init__(self):
        super(Model, self).__init__()
        self.conv1 = nn.Conv2d(3, 6, 5)
        self.conv2 = nn.Conv2d(6, 16, 5)
        self.fc1 = nn.Linear(16 * 13 * 13, 120)
        self.fc2 = nn.Linear(120, 84)
        self.fc3 = nn.Linear(84, 2)

    def forward(self, x):
        x = F.max_pool2d(F.relu(self.conv1(x)), 2)
        x = F.max_pool2d(F.relu(self.conv2(x)), 2)
        x = x.view(-1, 16 * 13 * 13)
        x = F.relu(self.fc1(x))
        x = F.relu(self.fc2(x))
        x = self.fc3(x)
        return x


model = Model()

# Define the loss
loss_fn = torch.nn.CrossEntropyLoss()

# Use the optim package to define an Optimizer that will update the weights of
# the model for us. Here we will use stochastic gredient descent; the optim package 
# contains many other optimization algoriths. The first argument to the SGD constructor tells the
# optimizer which Variables it should update.
learning_rate = 0.0005
optimizer = torch.optim.SGD(model.parameters(), learning_rate, momentum=0.9)

# Use to keep track of the cost
learn_cost = [] 

for t in range(2500):
    # Forward pass: compute predicted y by passing x to the model.
    o = model(torch.FloatTensor(X_train))

    # Compute and print loss.
    loss = loss_fn(o, torch.LongTensor(Y_train.reshape(-1,)))

    # Before the backward pass, use the optimizer object to zero all of the
    # gradients for the variables it will update (which are the learnable weights
    # of the model)
    optimizer.zero_grad()

    # Backward pass: compute gradient of the loss with respect to model
    # parameters
    loss.backward()

    # Calling the step function on an Optimizer makes an update to its
    # parameters
    optimizer.step()
    
    # Keep track of the cost
    learn_cost.append(loss.data)
    

# Plot the cost
plt.plot(np.squeeze(learn_cost))
plt.ylabel('cost')
plt.xlabel('iterations')
plt.title("Learning rate =" + str(learning_rate))
plt.show()
